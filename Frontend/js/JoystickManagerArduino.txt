if(PS4Controller.left_y > 0.1){
    throttle_control = commands_instance.throttle - parseInt((-PS4Controller.left_y * 450) + NEUTRAL_THROTTLE)
    if(throttle_control > limit_control){
        commands_instance.throttle -= limit_control
    }
    else if ( commands_instance.throttle < MIN_VEL){
        commands_instance.throttle = MIN_VEL
    }else{
        commands_instance.throttle = parseInt((-PS4Controller.left_y * 450) + NEUTRAL_THROTTLE)
    }
}
else if(PS4Controller.left_y < -0.1){
    throttle_control =  parseInt((-PS4Controller.left_y * 450) + NEUTRAL_THROTTLE) -  commands_instance.throttle
    if(throttle_control > limit_control){
        commands_instance.throttle += limit_control
    }
    else if ( commands_instance.throttle > MAX_VEL){
        commands_instance.throttle = MAX_VEL
    }else{
        commands_instance.throttle = parseInt((-PS4Controller.left_y * 450) + NEUTRAL_THROTTLE)
    }
   
}
else{
    commands_instance.throttle = NEUTRAL_THROTTLE;
}
